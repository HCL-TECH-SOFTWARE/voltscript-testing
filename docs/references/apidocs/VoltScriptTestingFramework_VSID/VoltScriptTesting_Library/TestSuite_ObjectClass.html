      • runAfterEach

        Public Sub runAfterEach()
        Tests whether a AbstractCustomBeforeAfter has been passed in and, if so, calls afterEach. Triggered from automatically all in-built assertions.
         
        Code Sample
        Dim globInt as Integer
        Class IntegerIncrementer as AbstractCustomBeforeAfter
            Sub beforeEach()
                a = a + 1
            End Sub
            Sub afterEach()
                Print "Finished test"
            End Sub
        End Class
        Private Class CustomTester as AbstractCustomTester
            Function runTests as Boolean
                Dim incrementer as New IntegerIncrementer()
                Call Me.testSuite.addCustomBeforeAfter(incrementer)
                Call Me.testSuite.checkStarted()
               
               Call Me.testSuite.runBeforeEach()   ' Increments a global variable
               Call Me.testSuite.describe("Test global variable").assertEqualsInteger(1, globInt)
               Call Me.testSuite.runAfterEach()
            End Function
        End Class
